# Composer P2 — Working Editor (enter · render · save · reload a melody) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Turn the Composer stub into a working single-staff melody editor — MIDI-pitch + numpad-duration note entry, live OSMD rendering, a caret, undo/delete, a per-user backend store with autosave + the data-loss validation gate, and a minimal gallery — so a kid can enter a melody, reload, and it survives.

**Architecture:** The tested document model (P1, done) drives the existing OSMD `MusicXmlRenderer` via `serializeFromEditor(state) → musicXml`. A React `EditorState` (owned by `EditorSurface` via `useState`) is mutated only through the model's `applyCommand(state, fn, ...)` history wrapper, dispatched by `useComposerInput` (numpad `keydown` + `usePianoMidi().subscribe` note-ons). A caret overlay positions itself from OSMD's `onLayout` step geometry mapped from the caret's model index. Autosave debounces, re-parses its own serialized XML (client validation gate) before `PUT`, and the backend `ComposerSongStore` persists `{id}.musicxml` + `{id}.meta.yml` (+ index cache + versions ring) under `data/users/{userId}/apps/piano/composer/`, re-validating on write.

**Tech Stack:** React 18 + the piano-kiosk mode framework (`usePianoMidi`, `usePianoKioskConfig`, `usePianoBreadcrumbBar`); OSMD via `MusicNotation/renderers/MusicXmlRenderer`; the Composer model barrel (`modes/Composer/model/index.js`); Node ESM backend (`YamlPianoStudioDatastore`/`PianoContainer` patterns, `FileIO`, `shared/music/musicXmlToNotes.mjs`); Vitest (colocated `*.test.{jsx,mjs}`, `piano.preset.test.mjs` idiom for routers) + one Playwright flow test.

## Global Constraints

- **P2 scope is the spec's P2 line** (`docs/reference/piano/composer.md` §14): EditorSurface + caret + sticky entry + auto-bar(split-tie, already in the model) + armed/disarmed; backend store (meta truth, versions ring, validation gate) + autosave; minimal gallery. Gate: enter + reload a melody end-to-end, data-loss test green against the live API.
- **P2 DEVIATION (deliberate, documented):** the spec's wet-ink `PendingLayer` requires a measure-capable renderer that does not exist (`SvgStaffRenderer` draws a single chord only; §15 lists this as unsized). P2 uses **engrave-per-edit** (re-serialize → OSMD engrave after each committed mutation) for correctness. The wet-ink `PendingLayer` + aged-page latency protocol (§2.1) is the FIRST P2-follow-up task, NOT dropped. Do not build a half-renderer under time pressure.
- **All mutations go through `applyCommand(state, fn, ...args)`** (history-aware; caret-only moves auto-skip history because they return the same `score` reference). Never mutate `EditorState` directly.
- **Model is DONE and frozen** — consume `modes/Composer/model/index.js`; do NOT modify the model in P2.
- **Logging:** child logger `piano-composer` via the framework (`getLogger().child({ component: 'piano-composer' })`), never raw `console.*`. Log the §12 events used in P2: `composer.mounted`, `composer.song.{created,opened,saved,save-rejected,save-invalid-xml}`, `composer.armed`/`composer.disarmed`, `composer.entry.note` (debug, sampled).
- **Backend ids are dot-free** (`/^[a-z0-9-]{1,64}$/`, server-generated via `shortId()`), because `saveYaml`/`deleteYaml` key off the trailing extension; the `.musicxml` blob is written with `writeBinary` (full filename), the `.meta.yml` with `saveYaml` (extension-less path).
- **Config accessor:** backend reads `configService.getHouseholdAppConfig(null, 'piano').composer`; the per-user dir is `path.join(configService.getUserDir(userId), 'apps', 'piano', 'composer')`. Frontend reads `usePianoKioskConfig().config.composer`.

```javascript
// backend/src/3_applications/piano/ComposerSongStore.mjs
import path from 'path';
import { loadYaml, saveYaml, listYamlFiles, deleteYaml, ensureDir, writeBinary, readFile, listFiles } from '#system/utils/FileIO.mjs';
import { shortId } from '#domains/core/utils/id.mjs';

const ID_RE = /^[a-z0-9-]{1,64}$/;

export class ComposerSongStore {
  #configService;
  #logger;
  constructor({ configService, logger = console }) {
    this.#configService = configService;
    this.#logger = logger;
  }

  isKnownUser(userId) {
    return typeof userId === 'string' && !userId.includes('/') && !userId.includes('..') && !!this.#configService.getUserProfile?.(userId);
  }
  #cfg() { return (this.#configService.getHouseholdAppConfig?.(null, 'piano') || {}).composer || {}; }
  #dir(userId) {
    if (!this.isKnownUser(userId)) return null;
    return path.join(this.#configService.getUserDir(userId), 'apps', 'piano', 'composer');
  }
  #metaPath(dir, id) { return path.join(dir, `${id}.meta`); }          // FileIO appends .yml
  #xmlPath(dir, id) { return path.join(dir, `${id}.musicxml`); }        // full filename via writeBinary/readFile
  #versionsDir(dir, id) { return path.join(dir, `${id}.versions`); }

  list(userId) {
    const dir = this.#dir(userId);
    if (!dir) return null;
    return listYamlFiles(dir)
      .filter(n => n.endsWith('.meta'))
      .map(n => loadYaml(path.join(dir, n)))
      .filter(Boolean)
      .map(m => ({ id: m.id, title: m.title, tags: m.tags || [], share: !!m.share, updatedAt: m.updatedAt, revision: m.revision }))
      .sort((a, b) => String(b.updatedAt).localeCompare(String(a.updatedAt)));
  }

  get(userId, id) {
    const dir = this.#dir(userId);
    if (!dir || !ID_RE.test(id)) return null;
    const meta = loadYaml(this.#metaPath(dir, id));
    if (!meta) return null;
    const musicxml = readFile(this.#xmlPath(dir, id));
    return { meta, musicxml };
  }

  create(userId, { title, musicxml, meta = {} } = {}) {
    const dir = this.#dir(userId);
    if (!dir) throw new Error('unknown user');
    ensureDir(dir);
    const id = shortId();
    const now = new Date().toISOString();
    const record = { id, title: title || `Song ${id}`, tags: meta.tags || [], share: !!meta.share, revision: 1, createdAt: now, updatedAt: now };
    writeBinary(this.#xmlPath(dir, id), musicxml || '');
    saveYaml(this.#metaPath(dir, id), record);
    this.#logger.info?.('composer.song.created', { userId, id });
    return record;
  }

  save(userId, id, { musicxml, meta = {}, revision } = {}) {
    const dir = this.#dir(userId);
    if (!dir || !ID_RE.test(id)) return { conflict: true, current: null };
    const cur = loadYaml(this.#metaPath(dir, id));
    if (!cur) return { conflict: true, current: null };
    if (typeof revision === 'number' && revision !== cur.revision) {
      this.#logger.info?.('composer.song.save-rejected', { userId, id, client: revision, server: cur.revision });
      return { conflict: true, current: cur };
    }
    // rotate the prior xml into the versions ring
    const vdir = this.#versionsDir(dir, id);
    ensureDir(vdir);
    const priorXml = readFile(this.#xmlPath(dir, id));
    if (priorXml != null) {
      writeBinary(path.join(vdir, `${cur.revision}.musicxml`), priorXml);
      const keep = Number(this.#cfg().versions_keep) || 5;
      const versions = listFiles(vdir).filter(n => n.endsWith('.musicxml'))
        .map(n => Number(n.replace('.musicxml', ''))).filter(Number.isFinite).sort((a, b) => a - b);
      while (versions.length > keep) { deleteYaml(path.join(vdir, `${versions.shift()}.musicxml`)); }
    }
    const now = new Date().toISOString();
    const next = { ...cur, title: meta.title ?? cur.title, tags: meta.tags ?? cur.tags, share: meta.share ?? cur.share, revision: cur.revision + 1, updatedAt: now };
    writeBinary(this.#xmlPath(dir, id), musicxml || '');
    saveYaml(this.#metaPath(dir, id), next);
    this.#logger.info?.('composer.song.saved', { userId, id, revision: next.revision });
    return { ok: true, revision: next.revision };
  }

  remove(userId, id) {
    const dir = this.#dir(userId);
    if (!dir || !ID_RE.test(id)) return false;
    deleteYaml(this.#metaPath(dir, id));
    deleteYaml(this.#xmlPath(dir, id));
    return true;
  }

  listVersions(userId, id) {
    const dir = this.#dir(userId);
    if (!dir) return [];
    return listFiles(this.#versionsDir(dir, id)).filter(n => n.endsWith('.musicxml'));
  }

  listShared() {
    const out = [];
    for (const u of (this.#configService.listHouseholdUsers?.() || this.#rosterFallback())) {
      const dir = this.#dir(u);
      if (!dir) continue;
      for (const n of listYamlFiles(dir).filter(x => x.endsWith('.meta'))) {
        const m = loadYaml(path.join(dir, n));
        if (m && m.share) out.push({ userId: u, id: m.id, title: m.title, tags: m.tags || [] });
      }
    }
    return out;
  }
  #rosterFallback() {
    // Test/dev: derive from any user dir we can reach. In prod listHouseholdUsers exists.
    return ['kc', 'soren'].filter(u => this.isKnownUser(u));
  }
}
```

> Note: `remove` uses `deleteYaml` for both the meta and the xml blob for test simplicity (the vi.mock deletes from either map by key membership). In prod, `writeBinary`/`readFile` handle the `.musicxml` full filename; a `deleteFile` helper exists in FileIO if `deleteYaml` refuses the non-.yml path — verify against `FileIO.mjs` and use the file-deleting helper for the blob if needed.

```javascript
      logger.info?.('composer.song.save-invalid-xml', { userId: req.params.userId, id: req.params.id });
      return res.status(400).json({ error: 'musicxml failed validation' });

    const r = cs.save(req.params.userId, req.params.id, { musicxml, meta, revision });

    res.json({ ok: cs.remove(req.params.userId, req.params.id), id: req.params.id });
  }));

  router.get('/compositions/shared', asyncHandler((req, res) => {
    res.json({ compositions: cs.listShared() });
  }));
```

```javascript
// frontend/src/modules/Piano/PianoKiosk/modes/Composer/useComposerInput.js
import { useEffect, useRef, useState } from 'react';
import { applyCommand, insertNote, insertRest, deleteNote, toggleDot, midiToPitch, moveCaret } from './model/index.js';

const DURATION_KEYS = { Numpad1: '16th', Numpad3: 'eighth', Numpad5: 'quarter', Numpad7: 'half', Numpad9: 'whole' };

export function mapKey(code) {
  if (DURATION_KEYS[code]) return { kind: 'duration', type: DURATION_KEYS[code] };
  switch (code) {
    case 'Numpad4': return { kind: 'arm' };
    case 'Numpad0': return { kind: 'rest' };
    case 'NumpadDecimal': return { kind: 'dot' };
    case 'NumpadSubtract': return { kind: 'deleteBack' };
    case 'Delete': return { kind: 'deleteAt' };
    case 'ArrowLeft': return { kind: 'caret', where: 'left' };
    case 'ArrowRight': return { kind: 'caret', where: 'right' };
    case 'PageUp': return { kind: 'caret', where: 'prevBar' };
    case 'PageDown': return { kind: 'caret', where: 'nextBar' };
    default: return null;
  }
}

export function useComposerInput({ setEditorState, subscribe }) {
  // Sticky entry state lives in a ref (read by the MIDI callback) + mirrored to React for the HUD.
  const sticky = useRef({ type: 'quarter', dots: 0, triplet: false });
  const armedRef = useRef(false);
  const [hud, setHud] = useState({ ...sticky.current, armed: false });
  const sync = () => setHud({ ...sticky.current, armed: armedRef.current });

  useEffect(() => {
    const onKey = (e) => {
      const m = mapKey(e.code);
      if (!m) return;
      e.preventDefault();
      switch (m.kind) {
        case 'duration': sticky.current = { ...sticky.current, type: m.type }; sync(); break;
        case 'dot': sticky.current = { ...sticky.current, dots: sticky.current.dots ? 0 : 1 }; sync(); break;
        case 'arm': armedRef.current = !armedRef.current; sync(); break;
        case 'rest': setEditorState((s) => applyCommand(s, insertRest, { ...sticky.current })); break;
        case 'deleteBack':
        case 'deleteAt': setEditorState((s) => applyCommand(s, deleteNote, s.caret)); break;
        case 'caret': setEditorState((s) => applyCommand(s, moveCaret, m.where)); break;
        default: break;
      }
    };
    window.addEventListener('keydown', onKey);
    return () => window.removeEventListener('keydown', onKey);
  }, [setEditorState]);

  useEffect(() => {
    if (!subscribe) return undefined;
    return subscribe((evt) => {
      if (!evt || evt.type !== 'note_on' || !evt.velocity) return;
      if (!armedRef.current) return; // disarmed = audition-only, no edit
      const pitch = midiToPitch(evt.note);
      setEditorState((s) => applyCommand(s, insertNote, pitch, { ...sticky.current }));
    });
  }, [subscribe, setEditorState]);

  return { hud, armed: hud.armed };
}
```

```jsx
export function CaretLayer({ steps = [], caretStepIndex = 0, scale = 1 }) {
  if (!steps.length) return null;
  const clamped = Math.min(caretStepIndex, steps.length - 1);
  const step = steps[clamped];
  const note = step?.notes?.[0];
  if (!note) return null;
  // Past-the-end caret parks just right of the last note; otherwise centers on the step.
  const past = caretStepIndex >= steps.length;
  const x = past ? note.x + (note.width || 12) + 6 * scale : note.x - 9 * scale;
  const height = Math.max(40 * scale, (note.bottom - note.top) || 40);
  return (
    <div
      className="composer-caret"
      style={{ transform: `translate3d(${x}px, ${note.top}px, 0)`, width: Math.round(18 * scale), height }}
    />
  );
}
```

```javascript
// frontend/src/modules/Piano/PianoKiosk/modes/Composer/useAutosave.js
import { useEffect, useRef, useState, useCallback } from 'react';
import { serializeFromEditor, parseMusicXml } from './model/index.js';

export function useAutosave({ editorState, id, revision, save, meta, idleMs = 3000 }) {
  const [status, setStatus] = useState('idle');
  const timer = useRef(null);
  const revRef = useRef(revision);
  useEffect(() => { revRef.current = revision; }, [revision]);

  const doSave = useCallback(async () => {
    if (!id) return;
    let xml;
    try {
      xml = serializeFromEditor(editorState);
      parseMusicXml(xml); // client validation gate — throws on bad xml
    } catch {
      setStatus('invalid');   // block the save, keep last-good
      return;
    }
    setStatus('saving');
    try {
      const r = await save(id, { musicxml: xml, meta, revision: revRef.current });
      if (r && r.ok) { revRef.current = r.revision; setStatus('saved'); }
      else setStatus('error');
    } catch { setStatus('error'); }
  }, [editorState, id, save, meta]);

  useEffect(() => {
    if (!editorState?.dirty) return undefined;
    clearTimeout(timer.current);
    timer.current = setTimeout(doSave, idleMs);
    return () => clearTimeout(timer.current);
  }, [editorState, idleMs, doSave]);

  const flush = useCallback(() => { clearTimeout(timer.current); doSave(); }, [doSave]);
  return { status, flush };
}
```

```jsx
// frontend/src/modules/Piano/PianoKiosk/modes/Composer/EditorSurface.jsx
import { useEffect, useMemo, useState, useCallback } from 'react';
import { MusicXmlRenderer } from '../../../../../MusicNotation/renderers/MusicXmlRenderer.jsx';
import { usePianoMidi } from '../../PianoMidiContext.jsx';
import getLogger from '../../../../../lib/logging/Logger.js';
import { initEditor, serializeFromEditor } from './model/index.js';
import { useComposerInput } from './useComposerInput.js';
import { useAutosave } from './useAutosave.js';
import { CaretLayer } from './CaretLayer.jsx';
import { StickyDurationHud } from './StickyDurationHud.jsx';

const logger = () => getLogger().child({ component: 'piano-composer' });

// Caret model position → engraved step index: count notes before the caret's measure + noteIdx.
function caretStepIndex(score, caret) {
  const measures = score?.parts?.[0]?.measures || [];
  let idx = 0;
  for (let m = 0; m < caret.measureIdx; m++) idx += (measures[m]?.notes?.length || 0);
  return idx + caret.noteIdx;
}

export function EditorSurface({ initialScore, songId, initialRevision = 1, save, config = {} }) {
  const [editorState, setEditorState] = useState(() => initEditor(initialScore));
  const [steps, setSteps] = useState([]);
  const { subscribe } = usePianoMidi();
  const { hud } = useComposerInput({ setEditorState, subscribe });
  useAutosave({ editorState, id: songId, revision: initialRevision, save, idleMs: config.autosave_idle_ms || 3000 });

  useEffect(() => { logger().info('composer.mounted', { songId }); }, [songId]);

  const musicXml = useMemo(() => serializeFromEditor(editorState), [editorState]);
  const onLayout = useCallback((res) => { setSteps(res?.steps || []); }, []);
  const stepIdx = caretStepIndex(editorState.score, editorState.caret);

  return (
    <div className="composer-editor">
      <StickyDurationHud hud={hud} />
      <MusicXmlRenderer musicXml={musicXml} flow="wrapped" scale={1} onLayout={onLayout}>
        <CaretLayer steps={steps} caretStepIndex={stepIdx} scale={1} />
      </MusicXmlRenderer>
    </div>
  );
}
```
